# 非运算

  

```
import cv2 as cv
import numpy as np

#  创建图片
img = np.zeros((200, 200), np.uint8)
#  设置区域颜色
img[50:150, 50:150] = 255
#  非运算
new_img = cv.bitwise_not(img)

#  显示图片
cv.imshow("img", img)
cv.imshow("new_img", new_img)
cv.waitKey(0)

```



# 与运算

```
import cv2 as cv
import numpy as np

#  创建图片
img = np.zeros((200, 200), np.uint8)
img2 = np.zeros((200, 200), np.uint8)
#  设置区域颜色
img[50:150, 50:150] = 255
img2[80:180, 80:180] = 255
#  与运算 重叠部分 
new_img = cv.bitwise_and(img, img2)

#  显示图片
cv.imshow("img", img)
cv.imshow("img2", img2)
cv.imshow("new_img", new_img)
cv.waitKey(0)

```
