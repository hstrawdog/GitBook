2. 样式

2.1. 样式1

{
    "title": "gitbook tutorial",
    "structure": {
        "readme": "README.md"
    },
    "plugins": [ 
        "-search",
        "github",
        "hide-element",
        "splitter", 
        "expandable-chapters",
        "tbfed-pagefooter",
        "back-to-top-button",
        "code",
        "page-toc-button" 
    ],
    "pluginsConfig": {
        "code": {
        "copyButtons": true
        },
          "hide-element": {
            "elements": [".gitbook-link"]
        },
        "github": {
            "url": "https://github.com/huangqiqiang/Book"
        },
        "tbfed-pagefooter": {
            "copyright":"Copyright &copy huangqiqiang 2020",
            "modify_label": "该文件修订时间：",
            "modify_format": "YYYY-MM-DD HH:mm:ss"
        },
        "theme-default": {
            "showLevel": true
        },
        "page-toc-button": {
            "maxTocDepth": 2,
            "minTocSize": 2
        }
    }
}
